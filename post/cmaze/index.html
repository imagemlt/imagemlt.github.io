templateclass pointStack {
T vendor;
int top;
public:
pointStack() {
vendor = new T[MAXSIZE];
top = -1;
}
void push(T p) {
vendor[++top] = p;
}
T pop() {
if (top == -1) { cout << “empty”; exit(0); }
return vendor[top–];
}
point operator[](int index) {
return T[index];
}
bool IsIn(T q) {
for (int i = 0; i <= top; i++) {
if (vendor[i] == q) {
return true;
}
}
return false;
}
int getTop() {
return top;
}
};
